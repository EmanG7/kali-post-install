function getip(){ 
    tunnel_ip=`ifconfig tun0 2>/dev/null | grep netmask | awk '{print $2}'` 
    default_ip=`ifconfig eth0 2>/dev/null | grep netmask | awk '{print $2}'` 
    if [[ $tunnel_ip == *"10."* ]]; then 
        echo $tunnel_ip 
    else 
        echo $default_ip 
    fi 
}
